Dependencies

  • AngularJS

Implementation

    <alert ng-model="$root.msg" ng-confirm="$root.confirm" ng-show="$root.msg.show" />

Factory

    'use strict';

    angular.module('example').factory('MsgFactory', function ($rootScope, $cookies) {
        return function () {
            this.msg = [];
            this.type = undefined;
            this.show = false;

            this.setMsg = function (value) {
                this.msg = value;
            };

            this.setType = function (value) {
                this.type = value;
            };

            this.setShow = function (value) {
                this.show = value;
            };

            this.push = function (value) {
                this.msg.push(value);
            };

            this.apply = function () {
                $rootScope.msg = this;
            };
        };
    });

Use Factory

    var msg = new MsgFactory();
    msg.setMsg('Informed message');
    msg.setType('confirm');
    msg.setShow(true);
    msg.apply();
